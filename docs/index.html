What's included

Everything your agent CI needs

From SDK-level recording to AGENTS.md generation — one package, zero configuration.

🎙️

SDK-Level Recording

Intercepts openai.chat.completions.create, anthropic.messages.create, tool calls, and agent decisions — not raw HTTP. Understands semantics, not just bytes.

v0.1
▶️

Deterministic Replay

Replay recorded traces with zero API calls and millisecond execution. Fully deterministic — perfect for CI pipelines and cost-free regression suites.

v0.1
🔍

Semantic Diff Engine

Compare traces at the trajectory level — detect model changes, new tools, extra LLM calls, and more. Binary match/no-match isn't enough for agents.

v0.1
🧪

pytest Plugin

Use the cassette fixture — auto-records on first run, replays on subsequent runs. Supports --record and --record-mode=all.

v0.1
💸

Budget Assertions

Guard against cost overruns, token bloat, and infinite tool loops with assert_cost_under, assert_no_loops, and the @budget pytest marker.

v0.2
⏱️

Time-Travel Debugger

Step forward and backward through a recorded trace. Inspect prompts, responses, tool I/O, and costs at each step from the terminal.

v0.2
📊

HTML Reports

Generate shareable single-file HTML reports with dark theme, stats grid, trajectory visualization, and expandable event panels.

v0.2
🔗

Async & Streaming

First-class support for async/await workflows and streaming responses. Streaming calls are assembled into complete cassettes and replayed as realistic chunks.

v0.2
🧠

LangGraph & CrewAI

Intercept Pregel graph execution, LangChain BaseChatModel calls, and CrewAI task pipelines. Capture graph-level and framework-level events.

v0.3
📚

RAG Recording & Scoring

Record retrieval queries, chunks, and vector store events. Assert on chunk count, relevance scores, and retrieval drift. Integrates with ragas and DeepEval.

v0.5
🔬

Semantic Regression

Catch meaning-level regressions that exact-string diffs miss. assert_semantic_similarity fails if any response drifts too far from the recorded baseline.

v0.5
🔌

MCP Tool Recording

Record Model Context Protocol server connections, tool calls, and results as first-class trace events. Full replay support for MCP-powered agents.

v0.5
🎓

Fine-Tune Export

Export recorded traces as OpenAI or Anthropic fine-tuning JSONL datasets directly from your cassette library — turn production runs into training data.

v0.5
📈

Pattern Detector

Analyze tool n-gram heatmaps, model usage stats, and error rates across an entire cassette directory. Surface how your agent actually navigates tasks.

v0.6 NEW
🔴

Gap Analyzer

Compare agent traces to golden baselines. Auto-detects token inflation, cost inflation, missing tools, model mismatch, error rate spikes. CI-exits with code 1 on critical gaps.

v0.6 NEW
📝

AGENTS.md Generator

Auto-generate AGENTS.md / CLAUDE.md style steering guidance from gap or pattern reports. Give your next agent run better instructions based on observed failures.

GitHub PR Fetcher

record_replay.py
from trace_ops import Recorder, Replayer, load_cassette

# ── Record ──────────────────────────────────────────────
with Recorder(save_to="cassettes/test_math.yaml") as rec:
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user", "content": "What is 2+2?"}],
    )
    print(response.choices[0].message.content)

print(f"Recorded {rec.trace.total_llm_calls} LLM calls")
print(f"Cost: ${rec.trace.total_cost_usd:.4f}")
print(f"Tokens: {rec.trace.total_tokens}")

# ── Replay — zero API calls, millisecond execution ──────
with Replayer("cassettes/test_math.yaml"):
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user", "content": "What is 2+2?"}],
    )
    assert response.choices[0].message.content == "4"   # deterministic!

# ── Async & Streaming ───────────────────────────────────
async with Recorder(save_to="cassettes/stream.yaml"):
    async for chunk in await client.chat.completions.create(
        model="gpt-4o", messages=[...], stream=True
    ):
        print(chunk.choices[0].delta.content or "", end="")
test_agent.py
from trace_ops import assert_trace_unchanged, load_cassette

# ── cassette fixture: auto-records on first run, replays after ──
def test_summarize(cassette):
    result = agent.run("Summarize the quarterly report")
    assert "revenue" in result.lower()

# ── Manual regression check ──────────────────────────────
def test_regression():
    baseline = load_cassette("cassettes/v1.yaml")
    with Recorder() as rec:
        agent.run("Summarize the quarterly report")
    assert_trace_unchanged(baseline, rec.trace)

# ── Diff and inspect ────────────────────────────────────
from trace_ops import diff_traces

def test_no_new_tools():
    old = load_cassette("cassettes/v1.yaml")
    new = load_cassette("cassettes/v2.yaml")
    diff = diff_traces(old, new)
    assert not diff.new_tools_used, f"Unexpected tools: {diff.new_tools_used}"
test_budget.py
import pytest
from trace_ops import Recorder
from trace_ops.assertions import (
    assert_cost_under,
    assert_tokens_under,
    assert_max_llm_calls,
    assert_no_loops,
)

def test_budget_manual():
    with Recorder() as rec:
        agent.run("Analyze this document")

    assert_cost_under(rec.trace, max_usd=0.50)
    assert_tokens_under(rec.trace, max_tokens=10_000)
    assert_max_llm_calls(rec.trace, max_calls=5)
    assert_no_loops(rec.trace, max_consecutive_same_tool=3)

# ── pytest marker shorthand ──────────────────────────────
@pytest.mark.budget(max_usd=0.50, max_tokens=10_000, max_llm_calls=5)
def test_agent_budget(cassette):
    agent.run("Analyze this document")
test_rag.py
from trace_ops import Recorder, load_cassette
from trace_ops.rag import (
    assert_chunk_count,
    assert_min_relevance_score,
    assert_no_retrieval_drift,
    RagasScorer,
)

def test_rag_pipeline():
    with Recorder(save_to="cassettes/rag_test.yaml") as rec:
        answer = rag_pipeline.run("What is the refund policy?")

    # Assert retrieval quality
    assert_chunk_count(rec.trace, min_chunks=3)
    assert_min_relevance_score(rec.trace, min_score=0.75)

def test_no_retrieval_drift():
    baseline = load_cassette("cassettes/rag_baseline.yaml")
    with Recorder() as rec:
        rag_pipeline.run("What is the refund policy?")
    assert_no_retrieval_drift(baseline, rec.trace, tolerance=0.1)

def test_ragas_scores():
    with Recorder() as rec:
        rag_pipeline.run("Explain the pricing tiers")
    scores = RagasScorer().score(rec.trace)
    assert scores["faithfulness"] > 0.85
    assert scores["answer_relevancy"] > 0.80
pattern_analysis.py
from trace_ops.analysis import PatternDetector, SkillsGenerator

# Analyze all cassettes in a directory
detector = PatternDetector(window_size=3, top_n=10)
report = detector.analyze_dir("cassettes/production/")

print(report.summary())
# Analyzed 47 traces | Avg: 3.2 LLM calls, 1,450 tokens, $0.012/run

print("\nTop tool sequences:")
for seq in report.top_tool_sequences:
    print(f"  {' → '.join(seq.sequence):<35} ×{seq.count}")
# search → read_file → write_file          ×31
# search → read_file                       ×9

print("\nModel usage:")
for m in report.model_usage:
    print(f"  {m.model}: {m.call_count} calls, {m.total_tokens} tokens")

# Auto-generate a pattern summary doc
gen = SkillsGenerator()
gen.from_pattern_report(report, output_path="PATTERNS.md")
gap_analysis.py
from trace_ops import load_cassette
from trace_ops.analysis import GapAnalyzer, SkillsGenerator

# Load golden (human-validated) and agent trace sets
golden = [(p.name, load_cassette(p)) for p in Path("golden/").glob("*.yaml")]
agent  = [(p.name, load_cassette(p)) for p in Path("runs/").glob("*.yaml")]

report = GapAnalyzer().compare(golden, agent)
print(report.summary())
# Found 3 behavioral gap(s): 1 critical, 2 warnings

for gap in report.gaps:
    icon = {"critical": "🔴", "warning": "🟡", "info": "🔵"}[gap.severity]
    print(f"{icon} [{gap.severity.upper()}] {gap.description}")
    if gap.golden_value is not None:
        print(f"   Golden: {gap.golden_value}  |  Agent: {gap.agent_value}")

# Gaps detected automatically:
# token_inflation, cost_inflation, missing_tool, extra_tool,
# model_mismatch, error_rate, llm_call_inflation

# Generate AGENTS.md steering guidance from the gaps
gen = SkillsGenerator()
gen.from_gap_report(report, output_path="AGENTS.md")

# CI-friendly: exits with code 1 if any critical gaps found

# Reverse-engineer a plain-English task from the PR diff

# Files changed: agent/pipeline.py, tests/test_pipeline.py
